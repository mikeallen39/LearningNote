```python
# layers/linear.py:277-319
class ColumnParallelLinear(LinearBase):
    """权重沿输出维度切分的线性层

    Y = XA + b, A = [A_1, ..., A_p]
    每个 GPU 持有 A_i，输出 Y_i = XA_i
    """

    def __init__(
        self,
        input_size: int,
        output_size: int,
        bias: bool = True,
        gather_output: bool = False,        # 是否 All-Gather 输出
        skip_bias_add: bool = False,
        params_dtype: Optional[torch.dtype] = None,
        quant_config: Optional[QuantizationConfig] = None,
        output_sizes: Optional[List[int]] = None,
        prefix: str = "",
        tp_rank: Optional[int] = None,       # 可选：指定 TP rank
        tp_size: Optional[int] = None,       # 可选：指定 TP size
        use_presharded_weights: bool = False,
        skip_block_quant_check: bool = False,
    ):
        super().__init__(...)

        self.gather_output = gather_output
        self.use_presharded_weights = use_presharded_weights

        # 获取 TP 信息
        if tp_rank is None:
            tp_rank = get_tensor_model_parallel_rank()
        if tp_size is None:
            tp_size = get_tensor_model_parallel_world_size()
        self.tp_rank, self.tp_size = tp_rank, tp_size

        # 计算分片大小
        self.output_size_per_partition = divide(self.output_size, tp_size)
```

### 1.3 Forward 实现

```python
# layers/linear.py:446-458
def forward(self, input_):
    bias = self.bias if not self.skip_bias_add else None

    # 矩阵乘法
    assert self.quant_method is not None
    output_parallel = self.quant_method.apply(self, input_, bias)

    if self.gather_output:
        # All-Gather 收集完整输出
        output = tensor_model_parallel_all_gather(output_parallel)
    else:
        output = output_parallel

    output_bias = self.bias if self.skip_bias_add else None
    return output, output_bias
```

```python
# layers/linear.py (简化版)
class RowParallelLinear(LinearBase):
    """权重沿输入维度切分的线性层

    A = [A_1; ...; A_p] (纵向拼接)
    每个 GPU 持有 A_i，输入 x_i，输出 y_i = x_i @ A_i
    最终输出 = All-Reduce(y_1, ..., y_p)
    """

    def __init__(
        self,
        input_size: int,
        output_size: int,
        bias: bool = True,
        input_is_parallel: bool = True,      # 输入是否已分片
        reduce_results: bool = True,          # 是否执行 All-Reduce
        skip_bias_add: bool = False,
        params_dtype: Optional[torch.dtype] = None,
        quant_config: Optional[QuantizationConfig] = None,
        prefix: str = "",
        tp_rank: Optional[int] = None,
        tp_size: Optional[int] = None,
    ):
        ...
        self.input_size_per_partition = divide(input_size, tp_size)
        self.reduce_results = reduce_results
```

### 2.3 Forward 实现

```python
def forward(self, input_, skip_all_reduce=False):
    bias = self.bias if not self.skip_bias_add else None

    # 矩阵乘法
    output_parallel = self.quant_method.apply(self, input_parallel, bias)

    # All-Reduce 合并结果
    if self.reduce_results and self.tp_size > 1 and not skip_all_reduce:
        output = tensor_model_parallel_all_reduce(output_parallel)
    else:
        output = output_parallel

    return output, output_bias
```

## 3. MergedColumnParallelLinear - 融合列并行

### 3.1 源码位置

`layers/linear.py:469-694`

### 3.2 类定义

```python
# layers/linear.py:469-500
class MergedColumnParallelLinear(ColumnParallelLinear):
    """融合多个列并行线性层

    将 gate_proj 和 up_proj 融合为单个矩阵
    权重: [input_size, output_size_1 + output_size_2 + ...]
    """

    def __init__(
        self,
        input_size: int,
        output_sizes: List[int],  # 每个融合层的输出大小
        bias: bool = True,
        gather_output: bool = False,
        skip_bias_add: bool = False,
        params_dtype: Optional[torch.dtype] = None,
        quant_config: Optional[QuantizationConfig] = None,
        prefix: str = "",
        tp_rank: Optional[int] = None,
        tp_size: Optional[int] = None,
        use_presharded_weights: bool = False,
    ):
        ...
        self.output_sizes = output_sizes
        # 计算每个输出的分片大小
        self.output_partition_sizes = [
            divide(output_size, tp_size) for output_size in output_sizes
        ]
```

### 3.3 优势

```
未融合:
├── gate_proj: [hidden, intermediate] → 1 次 GEMM
└── up_proj:   [hidden, intermediate] → 1 次 GEM
总计: 2 次 kernel 启动

融合后:
└── gate_up_proj: [hidden, intermediate * 2] → 1 次 GEM
总计: 1 次 kernel 启动
```
